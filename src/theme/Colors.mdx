import { Meta, ColorPalette, ColorItem } from '@storybook/blocks';
import { colors } from './colors'

<Meta title='Colors' />

# Colors

Developers should prefer using named-color classNames instead of hardcoding hex/rgba values. They are shown here for reference only.

<ColorPalette>
  {Object.entries(colors).map(([colorName, colorValue]) => {
    if(typeof colorValue === 'string') {
      return <ColorItem key={colorName} title={colorName} colors={{ [colorName]: colorValue }} />
    } else {
      return <ColorItem key={colorName} title={colorName} colors={colorValue} />
    }
  })}
</ColorPalette>
